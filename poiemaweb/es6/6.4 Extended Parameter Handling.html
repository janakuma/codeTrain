2. Rest 파라미터

2.1 기본 문법

rest 파라미터는 스프레드 연사자를 사용하여 파라미터를 정의한 것을 의미한다. rest 파라미터를 사용하면 인수의 리스트를 함수 내부에서 배열로 전달 받을 수 있다. 

function foo(...rest)	 {
	console.log(Array.isArray(rest)); //true
	console.log(rest);
}
foo(1,2,3,4,5);

function foo(...rest) {
	console.log(Array.isArray(rest));
	console.log(rest);
	return rest;
}
foo(1,2,3,4,5);
rest.push(10);
